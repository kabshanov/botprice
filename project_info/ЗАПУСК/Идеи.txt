## 3. –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –æ–±—Ä–∞–±–æ—Ç–∫–∏ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–π –≤ Telegram-–±–æ—Ç–∞
```python
from telegram import Update
from telegram.ext import Updater, CommandHandler, MessageHandler, Filters, CallbackContext
import pytesseract
from PIL import Image
import io

def image_handler(update: Update, context: CallbackContext):
    """–û–±—Ä–∞–±–∞—Ç—ã–≤–∞–µ—Ç –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—è –∏ —Ä–∞—Å–ø–æ–∑–Ω–∞–µ—Ç —Ç–µ–∫—Å—Ç."""
    file = update.message.photo[-1].get_file()  # –ü–æ–ª—É—á–∞–µ–º –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏–µ
    image_bytes = file.download_as_bytearray()  # –ó–∞–≥—Ä—É–∂–∞–µ–º –≤ –±–∞–π—Ç–∞—Ö
    image = Image.open(io.BytesIO(image_bytes))  # –û—Ç–∫—Ä—ã–≤–∞–µ–º —Å –ø–æ–º–æ—â—å—é PIL

    # –†–∞—Å–ø–æ–∑–Ω–∞–µ–º —Ç–µ–∫—Å—Ç
    text = pytesseract.image_to_string(image, lang="rus")

    update.message.reply_text(f"–†–∞—Å–ø–æ–∑–Ω–∞–Ω–Ω—ã–π —Ç–µ–∫—Å—Ç:\n{text}")

def main():
    """–ó–∞–ø—É—Å–∫ –±–æ—Ç–∞"""
    updater = Updater("YOUR_BOT_TOKEN", use_context=True)
    dp = updater.dispatcher
    dp.add_handler(MessageHandler(Filters.photo, image_handler))  # –û–±—Ä–∞–±–æ—Ç—á–∏–∫ —Ñ–æ—Ç–æ

    updater.start_polling()
    updater.idle()

if __name__ == "__main__":
    main()
```
